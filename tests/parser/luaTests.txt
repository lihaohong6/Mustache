		title = frame.args[1],

	return mw.ext.mustache.render("LuaSimpleTest", data)
end

!! article
Mustache:LuaArrayTest
!! text
<ul>{{#items}}<li>{{.}}</li>{{/items}}</ul>
!! endarticle

!! article
Module:LuaArrayTest
!! text
local p = {}

function p.render(frame)
	local items = {}
	for key, value in pairs(frame.args) do
		if string.sub(key, 1, 4) == "item" then
			table.insert(items, value)
		end
	end
	table.sort(items)
	local data = {
		items = items
	}
	return mw.ext.mustache.render("LuaArrayTest", data)
end

return p
!! endarticle

!! article
Module:LuaComplexTest
!! text
local p = {}

function p.render(frame)
	local data = {
		categories = {
			{
				name = "Electronics",
				items = {
					{name = "Laptop", price = 999, inStock = true},
					{name = "Phone", price = 599, inStock = false}
				}
			},
			{
				name = "Books",
				items = {
					{name = "Novel", price = 19, inStock = true}
				}
			}
		}
	}
	return mw.ext.mustache.render("LuaComplexTest", data)
end

return p
!! endarticle
