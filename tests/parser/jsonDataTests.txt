!! article
Module:JsonDataGenerator
!! text
local p = {}

function p.getCatalog()
    return [[{"categories":
    	[
    		{"name":"Electronics","items":[{"name":"Laptop","price":999,"inStock":true},{"name":"Phone","price":599,"inStock":false}]},
    		{"name":"Books","items":[{"name":"Novel","price":19,"inStock":true}]}
    	]
    }]]
end

return p
!! endarticle

!! article
Mustache:LuaDataTemplate
!! text
<div>
{{#catalog.categories}}
<h3>{{name}}</h3>
<ul>{{#items}}<li>{{name}} - ${{price}} ({{#inStock}}In stock{{/inStock}}{{^inStock}}Out of stock{{/inStock}})</li>{{/items}}</ul>
{{/catalog.categories}}
</div>
!! endarticle
