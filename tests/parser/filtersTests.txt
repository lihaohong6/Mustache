# ========================
# css-selector filter
# ========================

!! article
Mustache:CssSelectorInStyle
!! text
<style>
#{{id|css-selector}} {}
</style>
!! endarticle

!! test
css-selector filter in style context: generates valid CSS selector
!! wikitext
{{#mustache:CssSelectorInStyle|id=hello-world}}
!! html
<style>
#hello-world {}
</style>
!! end

!! test
css-selector filter in style context: strips spaces from CSS selector
!! wikitext
{{#mustache:CssSelectorInStyle|id=user name}}
!! html
<style>
#username {}
</style>
!! end

# ========================
# url filter
# ========================

!! article
Mustache:UrlFilter
!! text
<a href="{{link|url}}">click</a>
!! endarticle

!! test
url filter: passes through https URL
!! wikitext
{{#mustache:UrlFilter|link=https://example.com}}
!! html
<a href="https://example.com">click</a>
!! end

!! test
url filter: passes through http URL
!! wikitext
{{#mustache:UrlFilter|link=http://example.com}}
!! html
<a href="http://example.com">click</a>
!! end

!! test
url filter: blocks javascript scheme
!! wikitext
{{#mustache:UrlFilter|link=javascript:evil()}}
!! html
<a href="">click</a>
!! end

!! test
url filter: normalizes protocol-relative URL to https
!! wikitext
{{#mustache:UrlFilter|link=//example.com/path}}
!! html
<a href="https://example.com/path">click</a>
!! end

!! test
url filter: blocks invalid URL
!! wikitext
{{#mustache:UrlFilter|link=not a url}}
!! html
<a href="">click</a>
!! end

# ========================
# js-string filter
# ========================

!! article
Mustache:JsStringFilter
!! text
<p>{{value|js-string}}</p>
!! endarticle

!! test
js-string filter: JSON-encodes plain string
!! wikitext
{{#mustache:JsStringFilter|value=hello world}}
!! html
<p>'hello world'</p>
!! end

!! test
js-string filter: hex-encodes single quotes
!! wikitext
{{#mustache:JsStringFilter|value=say 'hi'}}
!! html
<p>'say \u0027hi\u0027'</p>
!! end

!! article
Mustache:JsStringInScript
!! text
<script>var msg = {{value|js-string}};</script>
!! endarticle

!! test
js-string filter in script context: encodes string value
!! wikitext
{{#mustache:JsStringInScript|value=hello world}}
!! html
<script>var msg = 'hello world';</script>
!! end

!! test
js-string filter in script context: hex-encodes script injection attempt
!! wikitext
{{#mustache:JsStringInScript|value=<script>}}
!! html
<script>var msg = '\u003Cscript\u003E';</script>
!! end
